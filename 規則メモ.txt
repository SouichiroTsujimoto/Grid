<...> <- 必須
[...] <- 必須ではない
{...} <- 繰り返し

[if式]
<ifex> <条件式> <:> <式> <:> <式>

# elseは必須 #

[if文] # 未実装 #

<if> <条件式> <do>
  <文>
[
<elif> <do>
  <文>
]
[
<else>
  <文>
<end>
]

[def文]
<def> <返り値の型> <関数名> <(> [ { <引数の型> <引数名> } ] <)> <do>
  <文>
  <return> <値>
<end>

[let文]
<let> <変数の型> <変数名> <=> <式>

[mut文]
<mut> <変数の型> <変数名> <=> <式>

[代入式]
<変数名> <:=> [ { <変数名> <:=> } ] <値>

# 代入できるのはmut文で定義された変数だけ #

[配列]
<\{> [ <値> { <,> <値> } ] <\}>

[map関数] # 後回し # 
<map> <(> <配列> <,> <関数> <)>

[for文]
<for> <型> <変数名> <<-> <配列> <do>
  <文>
<end>

[C++コード埋め込み] # 後で変更 #
<{> <C++のコード> <}>

# C++のコードの最後にはセミコロンを付けない #
