<...> <- 必須
[...] <- 必須ではない
{...} <- 繰り返し

[if式]
<if> <条件式> <do>
  <文>
[
<elif> <do>
  <文>
]
<else>
  <文>
<end>

# elseは必須 #

[def文]
<def> <返り値の型> <関数名> <(> [ { <引数の型> <引数名> } ] <)> <do>
  <文>
  <return> <値>
<end>

[let文]
<let> <変数の型> <変数名> <=> <式>

[C++コード埋め込み]
<{> <C++のコード> <}>

# C++のコードの最後にはセミコロンを付けない #
