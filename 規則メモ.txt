<...> <- 必須
[...] <- 必須ではない
{...} <- 繰り返し

#--実装済み--#

[if式]
<ifex> <条件式> <:> <式> <:> <式>

[if文]

<if> <条件式> <do>
  <文>
[
<elif> <do>
  <文>
]
[
<else>
  <文>
<end>
]

[def文]
<def> <返り値の型> <関数名> <(> [ { <引数の型> <引数名> } ] <)> <do>
  <文>
  <return> <値>
<end>

[let文]
<let> <変数の型> <変数名> <=> <式>

[var文]
<var> <変数の型> <変数名> <=> <式>

[代入式]
<変数名> <:=> [ { <変数名> <:=> } ] <値>

[配列]
<\{> [ <値> { <,> <値> } ] <\}>

[for文]
<for> <型> <変数名> <<-> <配列> <do>
  <文>
<end>

[パイプライン演算子]
<値> { < |> > <関数> [ { (引数) } ] }

# 値は関数の第一引数になる #
