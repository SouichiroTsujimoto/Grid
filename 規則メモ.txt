#--実装済み--#

[if式]
<ifex> <条件式> <:> <式> <:> <式>

[if文]

<if> <条件式> <do>
  <文>
[
<elif> <do>
  <文>
]
[
<else>
  <文>
<end>
]

[def文]
<def> <返り値の型> <関数名> <(> [ { <引数の型> <引数名> } ] <)> <do>
  <文>
  <return> <値>
<end>

[変数]
<型> <名前> <=> <値>

# 特別な場合を除いて初期化は必須 #
# 特別な場合を除いて再代入は禁止 #


[代入式]
<変数名> <:=> [ { <変数名> <:=> } ] <値>

[配列]
<\{> [ <値> { <,> <値> } ] <\}>

[for文]
<for> <型> <変数名> <<-> <配列> <do>
  <文>
<end>

[パイプライン演算子]
<値> { < |> > <関数> [ { (引数) } ] }

# 値は関数の第一引数になる #

[map関数]
<map> <(> <配列> <,> <関数> <)>

# 配列の要素は関数の第一引数になる #
# 第二引数の関数が返す値は、引数の配列の要素と同じ型でないとエラー #

[コメント]
</*> <文字列> <*/>

# 複数行に渡ってコメント化することができる #
# 単一行コメントは未実装 #
# コメント終了記号がない場合はエラー #

# (例) ka23 main.ka23 -ast #
# オプションは頭に'-'を付けないとエラー #
